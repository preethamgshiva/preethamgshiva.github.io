{

deno=2*a;
if(disc>0)
{
printf("Roots are real and distinct.\n");

root2=(-b+sqrt(disc))/deno;
printf("Root1=%f and Root2=%f",root1,root2);
}
else if(disc==0)
{
printf("Roots are real and distinct.\n");

}
else
{
printf("Roots are complex.\n");
