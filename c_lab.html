Swap using repetative subtraction.

//Swapping using repetative subtraction.
#include
int main()
{
int a,b;
printf("Enter two numbers");
scanf("%d%d",&a,&b);
printf("Numbers before swapping a=%d and b=%d\n",a,b);
a=a+b;
b=a-b;
a=a-b;
printf("Numbers after swapping a=%d and b=%d\n",a,b);
}
