#Program to find average of 5 integers.
num1=int(input("Enter Integer 1: "))
num2=int(input("Enter Integer 2: "))
num3=int(input("Enter Integer 3: "))
num4=int(input("Enter Integer 4: "))
num5=int(input("Enter Integer 5: "))

all_sum=num1+num2+num3+num4+num5
average=all_sum/5

print("Average of the 5 integers is ",average)
